  /**
   * @name calculateTime
   * @description Calculates time difference total for a list of time entry pairs
   * @method calculateTime
   * @memberof Consts
   * @param  {Date[]}      timeEntries List of time entry pairs
   * @return {Object}
   */
  static calculateTime(timeEntries){
    let hour = 0
    let min = 0
    for (let i = 0; i < timeEntries.length-1; i+=2){
      let start = timeEntries[i+1]
      let end = timeEntries[i]
      if (timeEntries[i+1].created){
        start = timeEntries[i+1].created
        end = timeEntries[i].created
      }
      let diff = Math.abs(new Date(start)-new Date(end))
      let diffHrs = Math.floor((diff % 86400000) / 3600000) // hour
      let diffMins = Math.round(((diff % 86400000) % 3600000) / 60000) // min
      hour += diffHrs
      if (min + diffMins > 60){
        hour += 1
        min += diffMins - 60
      }
      else {
        min += diffMins
      }
    }
    return {hour, min}
  }
